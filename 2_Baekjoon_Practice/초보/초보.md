### 1371

```python
import sys
sys.stdin = open("input.txt")

input = sys.stdin.read

sentence = input().replace("\n", "").replace(" ", "")
cnt = [0]*26
for i in sentence:
    cnt[ord(i)-97] += 1
max_cnt = max(cnt)
r = []
for i in range(len(cnt)):
    if cnt[i] == max_cnt:
        r.append(chr(i+97))
r.sort()
print(*r, sep="")
```

### 1547

```python
import sys
sys.stdin = open("input.txt")

n = int(input())

cups = [1, 2, 3]
for i in range(n):
    a, b = map(int, input().split())

    ai = cups.index(a)
    bi = cups.index(b)

    cups[ai], cups[bi] = cups[bi], cups[ai]
print(cups[0])
```

### 1568

```python
import sys
sys.stdin = open("input.txt")

n = int(input())

cnt = 0
k = 1

while n > 0:
    if n < k:
        k = 1
    n -= k
    k += 1
    cnt += 1
print(cnt)
```

n = int(input())

lst = []
for _ in range(n):
    lst.append(int(input()))

rlt = lst[-1]
if lst[2] - lst[1] == lst[1] - lst[0]:
